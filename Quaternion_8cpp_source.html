187 /*
188  Quaternion.matrix Matrix representation of Quaternion
189 
190  M = Q.matrix() is a matrix (4x4) representation of the Quaternion Q.
191 
192  Quaternion, or Hamilton, multiplication can be implemented as a
193  matrix-vector product, where the column-vector is the elements of a
194  second quaternion:
195 
196  matrix(Q1) * double(Q2)'
197 
198  Notes::
199  - This matrix is not unique, other matrices will serve the purpose for
200  multiplication, see https://en.wikipedia.org/wiki/Quaternion#Matrix_representations
201  - The determinant of the matrix is the norm of the quaternion to the fourth power.
202 
203  See also Quaternion.double, Quaternion.mtimes.
204 */
205 Matrix<4, 4> Quaternion::matrix() const
206 {
207  /*
208  Matlab Version:
209  m = [q.s -q.v(1) -q.v(2) -q.v(3)
210  q.v(1) q.s -q.v(3) q.v(2)
211  q.v(2) q.v(3) q.s -q.v(1)
212  q.v(3) -q.v(2) q.v(1) q.s];
213  */
214  return Data(_s, -_v[0], -_v[1], -_v[2], _v[0], _s, -_v[2], _v[1], _v[1], _v[2], _s, -_v[0], _v[2], -_v[1], _v[0], _s);
215 }

237 /*
238  Quaternion.mtimes Multiply a quaternion object
239 
240  Q1*Q2 is a quaternion formed by the Hamilton product of two quaternions.
241  Q*S is the element-wise multiplication of quaternion elements by the scalar S.
242  S*Q is the element-wise multiplication of quaternion elements by the scalar S.
243 
244  Notes::
245  - Overloaded operator '*'
246 
247  See also Quaternion.mrdivide, Quaternion.mpower.
248 */
249 Quaternion Quaternion::mtimes(const Quaternion& q2) const
250 {
251  double s1 = _s;
252  double s2 = q2.getS();
253  Vector<3> v1 = _v;
254  Vector<3> v2 = q2.getV();
255  return Quaternion(s1 * s2 - (v1 * v2), s1 * v2 + s2 * v1 + (v1 ^ v2));
256 }

258 {
259  return Quaternion(getDouble() * s);
260 }

298 /*
299  Quaternion.mpower Raise quaternion to integer power
300 
301  Q^N is the Quaternion Q raised to the integer power N.
302 
303  Notes::
304  - Overloaded operator '^'
305  - Computed by repeated multiplication.
306  - If the argument is a unit-quaternion, the result will be a
307  unit quaternion.
308 
309  See also Quaternion.mtimes.
310 */
311 Quaternion Quaternion::mpower(int p) const
312 {
313  if (p == 0)
314  return Quaternion(1.0, Zeros);
315 
316  // multiply by itself so many times
317  Quaternion q(*this);
318  for (int i = 1; i < ::abs(p); i++)
319  {
320  q = q * (*this);
321  }
322 
323  // if exponent was negative, invert it
324  if (p < 0)
325  {
326  q = sun::inv(q);
327  }
328 
329  return q;
330 }
331 Quaternion Quaternion::operator^(const int& p) const
332 {
333  return mpower(p);
334 }

402 /*
403  EQ Test quaternion equality
404 
405  Q1==Q2 is true if the quaternions Q1 and Q2 are equal.
406 
407  Notes::
408  - Overloaded operator '=='.
409  - This method is invoked for unit Quaternions where Q and -Q represent

411  equivalent.

414 */

416 {
417  return (sum(abs(getDouble() - q2.getDouble())) < 100.0 * GEOMETRY_HELPER_EPSILON);
418 }
