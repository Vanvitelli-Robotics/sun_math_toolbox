126 /*
127  Compute pinv as Damped Least Square (DLS)
128 */
129 Matrix<> pinv_DLS(const Matrix<>& M, double damping)
130 {
131  SVD<> M_svd(M);
132 
133  Matrix<> V = M_svd.get_VT().T();
134  Matrix<> UT = M_svd.get_U().T();
135  Vector<> Sigma = M_svd.get_diagonal();
136 
137  // Calc lambda
138  double eps = std::numeric_limits<double>::epsilon();
139  double sigma_min = Sigma[Sigma.size() - 1];
140  double lambda_square;
141  if (sigma_min >= damping)
142  {
143  lambda_square = eps;
144  }
145  else if (sigma_min >= damping / 2.0)
146  {
147  lambda_square = sigma_min * (damping - sigma_min) + eps;
148  }
149  else
150  {
151  lambda_square = pow(damping, 2) / 4.0 + eps;
152  }
153 
154  // Invert Sigma
155  Vector<> Sigma_dls = Zeros(Sigma.size());
156  for (int i = 0; i < Sigma_dls.size(); i++)
157  {
158  Sigma_dls[i] = Sigma[i] / (pow(Sigma[i], 2) + lambda_square);
159  }
160 
161  // Do V*Sigma_dls*UT = sum( Sigma_dls[i]*v[i]*u[i]T )
162  int v_num_rows = V.num_rows();
163  int ut_num_cols = UT.num_cols();
164  Matrix<> M_pinv = Sigma_dls[0] * V.slice(0, 0, v_num_rows, 1) * UT.slice(0, 0, 1, ut_num_cols);
165  for (int i = 1; i < Sigma.size(); i++)
166  {
167  M_pinv = M_pinv + Sigma_dls[i] * V.slice(0, i, v_num_rows, 1) * UT.slice(i, 0, 1, ut_num_cols);
168  }
169 
170  return M_pinv;
171 }
