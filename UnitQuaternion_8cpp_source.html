191 /*
192  UnitQuaternion.interp Interpolate UnitQuaternions
193 
194  QI = Q.scale(S, OPTIONS) is a UnitQuaternion that interpolates between a null
195  rotation (identity quaternion) for S=0 to Q for S=1.
196 
197  QI = Q.interp(Q2, S, OPTIONS) as above but interpolates a rotation
198  between Q for S=0 and Q2 for S=1.
199 
200  Options::
201  'shortest' Take the shortest path along the great circle
202 
203  Notes::
204  - This is a spherical linear interpolation (slerp) that can be interpretted
205  as interpolation along a great circle arc on a sphere.
206  - It is an error if S is outside the interval 0 to 1.
207 
208  References::
209  - Animating rotation with quaternion curves,
210  K. Shoemake,
211  in Proceedings of ACM SIGGRAPH, (San Francisco), pp. 245-254, 1985.
212 
213  See also ctraj.
214 */
215 UnitQuaternion UnitQuaternion::interp(const UnitQuaternion& q2, double s, bool b_shortest) const
216 {
217  Vector<4> q1_vec = getDouble();
218  Vector<4> q2_vec = q2.getDouble();
219 
220  double cosTheta = q1_vec * q2_vec;
221 
222  if (b_shortest)
223  {
224  // take shortest path along the great circle, patch by Gauthier Gras
225  if (cosTheta < 0)
226  {
227  q1_vec = -q1_vec;
228  cosTheta = -cosTheta;
229  }
230  }
231 
232  double theta = acos(cosTheta);
233 
234  // check s range
235  if (s < 0 || s > 1)
236  {
237  cout << PORTING_FUNCTIONS_FAIL_COLOR "[UnitQuaternion]: error in interp() | s must be in [0 1] | s = " << s
238  << PORTING_FUNCTIONS_CRESET << endl;
239  exit(-1);
240  }
241 
242  if (theta == 0)
243  return UnitQuaternion(q1_vec);
244 
245  return UnitQuaternion((sin((1 - s) * theta) * q1_vec + sin(s * theta) * q2_vec) / sin(theta));
246 }

290 /*
291  UnitQuaternion.dot Quaternion derivative
292 
293  QD = Q.dot(omega) is the rate of change in the world frame of a body
294  frame with attitude Q and angular velocity OMEGA (1x3) expressed as a
295  quaternion.
296 
297  Notes::
298  - This is not a group operator, but it is useful to have the result as a
299  quaternion.
300 
301  Reference::
302  - Robotics, Vision & Control, 2nd edition, Peter Corke, Chap 3.
303 
304  See also UnitQuaternion.dotb.
305 */
306 // The return type is not UnitQuaternion
307 Quaternion UnitQuaternion::dot(const Vector<3>& omega) const
308 {
309  // UnitQuaternion.pure(omega) * (*this)
310  Matrix<3, 3> E = _s * Identity(3) - skew(_v);
311  return Quaternion(-0.5 * _v * omega, 0.5 * E * omega);
312 }
313 
314 /*
315  UnitQuaternion.dotb Quaternion derivative IN BODY FRAME
316 
317  QD = Q.dotb(omega) is the rate of change in the BODY frame of a body frame
318  with attitude Q and angular velocity OMEGA (1x3) expressed as a
319  quaternion.
320 
321  Notes::
322  - This is not a group operator, but it is useful to have the result as a
323  quaternion.
324 
325  Reference::
326  - Robotics, Vision & Control, 2nd edition, Peter Corke, Chap 3.
327 
328  See also UnitQuaternion.dot.
329 */
330 Quaternion UnitQuaternion::dotb(const Vector<3>& omega) const
331 {
332  // q * UnitQuaternion.pure(omega)
333  Matrix<3, 3> E = _s * Identity(3) + skew(_v); // the difference is the PLUS!! +++
334  return Quaternion(-0.5 * _v * omega, 0.5 * E * omega);
335 }

593 /*
594  TOROT Convert UnitQuaternion to homogeneous transform
595 
596  T = q2tr(Q)
597 
598  Return the rotational homogeneous transform corresponding to the unit
599  quaternion Q.
600 
601  See also: TR2Q
602 */
603 Matrix<3, 3> UnitQuaternion::torot() const
604 {
605  Vector<4> q_vec = getDouble();
606  double s = q_vec[0];
607  double x = q_vec[1];
608  double y = q_vec[2];
609  double z = q_vec[3];
610 
611  return Data(
612 
613  1.0 - 2.0 * (pow(y, 2) + pow(z, 2)), 2.0 * (x * y - s * z), 2.0 * (x * z + s * y), 2.0 * (x * y + s * z),
614  1.0 - 2.0 * (pow(x, 2) + pow(z, 2)), 2.0 * (y * z - s * x), 2.0 * (x * z - s * y), 2.0 * (y * z + s * x),
615  1.0 - 2.0 * (pow(x, 2) + pow(y, 2))
616 
617  );
618 }

757 /*
758  TR2Q Convert homogeneous transform to a UnitQuaternion
759 
760  Q = tr2q(T)
761 
762  Return a UnitQuaternion corresponding to the rotational part of the
763  homogeneous transform T.
764 
765  Reference::
766  - Funda, Taylor, IEEE Trans. Robotics and Automation, 6(3), June 1990, pp.
767 */
768 UnitQuaternion UnitQuaternion::r2q(const Matrix<3>& R)
769 {
770  // Check matrix
771  if (!isRot(R))
772  {
773  throw std::invalid_argument("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): R is not a rotation Matrix");
774  }
775 
776  // Using modified version of sign() function as per the paper
777  // sign(x) = 1 if x>=0
778 
779  double trace_1 = trace(R) + 1.0;
780  if (trace_1 < 0.0)
781  {
782  if (trace_1 > -10.0 * std::numeric_limits<double>::epsilon())
783  {
784  trace_1 = 0.0; //<- Numeric fix
785  }
786  else
787  {
788  // trace+1 is negative for another reason
789  throw std::logic_error("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): trace+1 of R is negative. Please "
790  "contact the mantainer.");
791  }
792  }
793  double s = sqrt(trace_1) / 2.0;
794  double kx = R[2][1] - R[1][2]; // Oz - Ay
795  double ky = R[0][2] - R[2][0]; // Ax - Nz
796  double kz = R[1][0] - R[0][1]; // Ny - Ox
797 
798  // for the numerical case, deal with rotations by very small angles
799 
800  // equation (7)
801  int k = index_max(R.diagonal_slice());
802  double kx1, ky1, kz1;
803  double sgn;
804  switch (k)
805  {
806  case 0:
807  { // Nx dominates
808  kx1 = R[0][0] - R[1][1] - R[2][2] + 1.0; // Nx - Oy - Az + 1
809  ky1 = R[1][0] + R[0][1]; // Ny + Ox
810  kz1 = R[2][0] + R[0][2]; // Nz + Ax
811  sgn = sign(kx, 1);
812  break;
813  }
814  case 1:
815  { // Oy dominates
816  kx1 = R[1][0] + R[0][1]; // Ny + Ox
817  ky1 = R[1][1] - R[0][0] - R[2][2] + 1.0; // Oy - Nx - Az + 1
818  kz1 = R[2][1] + R[1][2]; // Oz + Ay
819  sgn = sign(ky, 1);
820  break;
821  }
822  case 2:
823  { // Az dominates
824  kx1 = R[2][0] + R[0][2]; // Nz + Ax
825  ky1 = R[2][1] + R[1][2]; // Oz + Ay
826  kz1 = R[2][2] - R[0][0] - R[1][1] + 1.0; // Az - Nx - Oy + 1
827  sgn = sign(kz, 1);
828  break;
829  }
830  default:
831  {
832  /* Default Code */
833  throw std::logic_error("[UnitQuaternion] Error in r2q( const Matrix<3,3>& R ): invalid index in index_max(). "
834  "Please contact the mantainer.");
835  }
836  }
837 
838  // equation (8)
839  kx = kx + sgn * kx1;
840  ky = ky + sgn * ky1;
841  kz = kz + sgn * kz1;
842 
843  Vector<3> k_vec = makeVector(kx, ky, kz);
844  double nm = TooN::norm(k_vec);
845  if (nm == 0)
846  {
847  // handle special case of null quaternion
848  return UnitQuaternion();
849  }
850  else
851  {
852  return UnitQuaternion(s, k_vec * sqrt(1.0 - pow(s, 2)) / nm);
853  }
854 }

860 
861  Return the rotational homogeneous transform corresponding to the unit
862  quaternion Q.

865 */
