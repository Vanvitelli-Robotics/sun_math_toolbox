117 Vector<3> r2rpy(const Matrix<3, 3> &R)
118 {
119  // Check matrix
120  if (!isRot(R))
121  {
122  throw std::invalid_argument("[PORTINGFUNCIONS] Error in r2rpy( const Matrix<3,3>& R ): R is not a rotation Matrix");
123  }
124 
125  // old ZYX order (as per Paul book)
126 
127  Vector<3> rpy = Zeros;
128 
129  if (fabs(fabs(R[2][0]) - 1.0) < GEOMETRY_HELPER_EPSILON)
130  { // when |R31| == 1
131  // singularity
132 
133  rpy[0] = 0; // roll is zero
134  if (R[2][0] < 0.0)
135  {
136  rpy[2] = -atan2(R[0][1], R[0][2]); // R-Y
137  }
138  else
139  {
140  rpy[2] = atan2(-R[0][1], -R[0][2]); // R+Y
141  }
142  rpy[1] = -asin(R[2][0]);
143  }
144  else
145  {
146  rpy[0] = atan2(R[2][1], R[2][2]); // R
147  rpy[2] = atan2(R[1][0], R[0][0]); // Y
148 
149  rpy[1] = -atan(R[2][0] * cos(rpy[0]) / R[2][2]);
150  }
151 
152  return rpy;
153 }

178 Vector<3> r2eul(const Matrix<3, 3> &R)
179 {
180  // Check matrix
181  if (!isRot(R))
182  {
183  throw std::invalid_argument("[PORTINGFUNCIONS] Error in r2eul( const Matrix<3,3>& R ): R is not a rotation Matrix");
184  }
185 
186  /* Method as per Paul, p 69.
187  euler = [phi theta psi]
188  */
189 
190  Vector<3> eul = Zeros;
191 
192  if (std::abs(R[0][2]) < GEOMETRY_HELPER_EPSILON && std::abs(R[1][2]) < GEOMETRY_HELPER_EPSILON)
193  {
194  // singularity
195  eul[0] = 0.0;
196  double sp = 0.0;
197  double cp = 0.0;
198  eul[1] = atan2(cp * R[0][2] + sp * R[1][2], R[2][2]);
199  eul[2] = atan2(-sp * R[0][0] + cp * R[1][0], -sp * R[0][1] + cp * R[1][1]);
200  }
201  else
202  {
203  // non singular
204  eul[0] = atan2(R[1][2], R[0][2]);
205  double sp = sin(eul[0]);
206  double cp = cos(eul[0]);
207  eul[1] = atan2(cp * R[0][2] + sp * R[1][2], R[2][2]);
208  eul[2] = atan2(-sp * R[0][0] + cp * R[1][0], -sp * R[0][1] + cp * R[1][1]);
209  }
210 
211  return eul;
212 }
